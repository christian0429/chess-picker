React Glider

A React wrapper for Glider.js

Usage

import Glider from 'react-glider';
import 'glider-js/glider.min.css';

function MyComponent() {
  return (
    <Glider
      draggable
      hasArrows
      hasDots
      slidesToShow={2}
      slidesToScroll={1}
    >
      <Pane>1</Pane>
      <Pane>2</Pane>
      <Pane>3</Pane>
      <Pane>4</Pane>
      <Pane>5</Pane>
    </Glider>
  );
}
